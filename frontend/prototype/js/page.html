function show(id) { forEach(id, function(id) { setStyle(getElement(id), 'display', ''); }); } function hide(id) { forEach(id, function(id) { setStyle(getElement(id), 'display', 'none'); }); } function forEach(args, fn) { if (typeof args == "object") { for (var i = 0, argLen = args.length; i < argLen; i++) { forEach(args[i], fn); } } else { fn(args); } } function setStyle(el, property, value) { el.style[property] = value; } function getElement(id) { return document.getElementById(id); }
